local GalaxyMoon = {}
GalaxyMoon.__index = GalaxyMoon

local TweenService = game:GetService("TweenService")
local UIS = game:GetService("UserInputService")
local Players = game:GetService("Players")
local lp = Players.LocalPlayer

local COL_ACCENT = Color3.fromRGB(210, 90, 255)
local COL_ACCENT2 = Color3.fromRGB(180, 0, 255)
local COL_BG = Color3.fromRGB(12, 12, 15)
local COL_BG2 = Color3.fromRGB(20, 20, 28)
local COL_BG3 = Color3.fromRGB(22, 18, 35)
local COL_OFF = Color3.fromRGB(180, 180, 180)

local dragTarget, dragStart, dragStartPos = nil, nil, nil

UIS.InputChanged:Connect(function(i)
	if not dragTarget then return end
	if i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch then
		local delta = i.Position - dragStart
		dragTarget.Position = UDim2.new(
			dragStartPos.X.Scale, dragStartPos.X.Offset + delta.X,
			dragStartPos.Y.Scale, dragStartPos.Y.Offset + delta.Y
		)
	end
end)

UIS.InputEnded:Connect(function(i)
	if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
		dragTarget = nil
	end
end)

local function makeDraggable(handle, target)
	handle.InputBegan:Connect(function(i)
		if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
			dragTarget = target
			dragStart = i.Position
			dragStartPos = target.Position
		end
	end)
end

local function animateText(label, text)
	label.Text = ""
	task.spawn(function()
		for i = 1, #text do
			label.Text = string.sub(text, 1, i)
			task.wait(0.04)
		end
	end)
end

local function corner(parent, radius)
	local c = Instance.new("UICorner", parent)
	c.CornerRadius = UDim.new(0, radius or 10)
	return c
end

local function wrap(parent, color, size, pos, radius)
	local f = Instance.new("Frame", parent)
	f.Size = size
	f.Position = pos
	f.BackgroundColor3 = color
	f.BorderSizePixel = 0
	corner(f, radius or 11)
	return f
end

local function innerFrame(border, bg, radius)
	local f = Instance.new("Frame", border)
	f.Size = UDim2.new(1, -2, 1, -2)
	f.Position = UDim2.new(0, 1, 0, 1)
	f.BackgroundColor3 = bg or COL_BG3
	f.BorderSizePixel = 0
	corner(f, radius or 10)
	return f
end

function GalaxyMoon.new(config)
	local self = setmetatable({}, GalaxyMoon)

	local name = config.Name or "GalaxyMoon"
	local W = config.Width or 320
	local FULL_H = config.Height or 280
	local MIN_H = 44
	local BORDER = 3
	local IMAGE_ID = config.Icon or ""

	if lp.PlayerGui:FindFirstChild("GalaxyMoon_" .. name) then
		lp.PlayerGui["GalaxyMoon_" .. name]:Destroy()
	end

	local gui = Instance.new("ScreenGui", lp.PlayerGui)
	gui.Name = "GalaxyMoon_" .. name
	gui.ResetOnSpawn = false
	gui.DisplayOrder = 10
	self.gui = gui

	local circleBorder = wrap(gui, COL_ACCENT, UDim2.new(0, 59, 0, 59), UDim2.new(0, 20, 0.5, -30), 100)
	circleBorder.Active = true

	local circleBtn = Instance.new("TextButton", circleBorder)
	circleBtn.Size = UDim2.new(1, -4, 1, -4)
	circleBtn.Position = UDim2.new(0, 2, 0, 2)
	circleBtn.Text = ""
	circleBtn.BackgroundColor3 = Color3.fromRGB(12, 12, 20)
	circleBtn.BorderSizePixel = 0
	corner(circleBtn, 100)

	if IMAGE_ID ~= "" then
		local img = Instance.new("ImageLabel", circleBtn)
		img.Size = UDim2.new(0.8, 0, 0.8, 0)
		img.Position = UDim2.new(0.1, 0, 0.1, 0)
		img.Image = IMAGE_ID
		img.BackgroundTransparency = 1
	else
		local lbl = Instance.new("TextLabel", circleBtn)
		lbl.Size = UDim2.new(1, 0, 1, 0)
		lbl.Text = "☽"
		lbl.Font = Enum.Font.GothamBold
		lbl.TextSize = 22
		lbl.TextColor3 = COL_ACCENT
		lbl.BackgroundTransparency = 1
	end

	makeDraggable(circleBorder, circleBorder)

	local borderFrame = wrap(gui, COL_ACCENT,
		UDim2.new(0, W + BORDER * 2, 0, FULL_H + BORDER * 2),
		UDim2.new(0.5, -(W / 2 + BORDER), 0.5, -(FULL_H / 2 + BORDER)), 16)
	borderFrame.Active = true
	borderFrame.Visible = false
	self.borderFrame = borderFrame

	local main = Instance.new("Frame", borderFrame)
	main.Size = UDim2.new(1, -BORDER * 2, 1, -BORDER * 2)
	main.Position = UDim2.new(0, BORDER, 0, BORDER)
	main.BackgroundColor3 = COL_BG
	main.BorderSizePixel = 0
	main.ClipsDescendants = true
	corner(main, 14)

	local topBar = Instance.new("Frame", main)
	topBar.Size = UDim2.new(1, 0, 0, 40)
	topBar.BackgroundColor3 = COL_BG2
	topBar.BorderSizePixel = 0
	corner(topBar, 0)
	makeDraggable(topBar, borderFrame)

	local titleLabel = Instance.new("TextLabel", topBar)
	titleLabel.Size = UDim2.new(1, -80, 1, 0)
	titleLabel.Position = UDim2.new(0, 15, 0, 0)
	titleLabel.Text = ""
	titleLabel.Font = Enum.Font.Arcade
	titleLabel.TextColor3 = Color3.new(1, 1, 1)
	titleLabel.TextSize = 14
	titleLabel.BackgroundTransparency = 1
	titleLabel.TextXAlignment = Enum.TextXAlignment.Left
	self.titleLabel = titleLabel

	local minBtn = Instance.new("TextButton", topBar)
	minBtn.Size = UDim2.new(0, 35, 0, 35)
	minBtn.Position = UDim2.new(1, -40, 0, 2.5)
	minBtn.Text = "-"
	minBtn.Font = Enum.Font.GothamBold
	minBtn.TextColor3 = Color3.new(1, 1, 1)
	minBtn.BackgroundTransparency = 1
	minBtn.TextSize = 25

	local tabBar = Instance.new("Frame", main)
	tabBar.Size = UDim2.new(1, 0, 0, 28)
	tabBar.Position = UDim2.new(0, 0, 0, 42)
	tabBar.BackgroundTransparency = 1
	local tabLayout = Instance.new("UIListLayout", tabBar)
	tabLayout.FillDirection = Enum.FillDirection.Horizontal
	tabLayout.Padding = UDim.new(0, 5)
	tabLayout.HorizontalAlignment = Enum.HorizontalAlignment.Left
	tabLayout.VerticalAlignment = Enum.VerticalAlignment.Center
	local tabPad = Instance.new("UIPadding", tabBar)
	tabPad.PaddingLeft = UDim.new(0, 10)
	self.tabBar = tabBar

	local contentArea = Instance.new("Frame", main)
	contentArea.Size = UDim2.new(1, 0, 1, -72)
	contentArea.Position = UDim2.new(0, 0, 0, 72)
	contentArea.BackgroundTransparency = 1
	self.contentArea = contentArea

	self.tabs = {}
	self.tabBtns = {}
	self.tabFrames = {}
	self.activeTab = nil

	local isMin = false
	minBtn.MouseButton1Click:Connect(function()
		isMin = not isMin
		minBtn.Text = isMin and "+" or "-"
		contentArea.Visible = not isMin
		tabBar.Visible = not isMin
		TweenService:Create(borderFrame, TweenInfo.new(0.4, Enum.EasingStyle.Quart), {
			Size = UDim2.new(0, W + BORDER * 2, 0, (isMin and MIN_H or FULL_H) + BORDER * 2)
		}):Play()
	end)

	circleBtn.MouseButton1Click:Connect(function()
		borderFrame.Visible = not borderFrame.Visible
		if borderFrame.Visible then
			animateText(titleLabel, name)
		end
	end)

	animateText(titleLabel, name)
	task.delay(0.5, function()
		borderFrame.Visible = true
	end)

	return self
end

function GalaxyMoon:CreateTab(tabName)
	local tabBtn = Instance.new("TextButton", self.tabBar)
	tabBtn.Size = UDim2.new(0, 80, 0, 22)
	tabBtn.Text = tabName
	tabBtn.Font = Enum.Font.Arcade
	tabBtn.TextSize = 10
	tabBtn.TextColor3 = Color3.new(1, 1, 1)
	tabBtn.BackgroundColor3 = (#self.tabs == 0) and COL_ACCENT2 or Color3.fromRGB(30, 30, 40)
	tabBtn.BorderSizePixel = 0
	corner(tabBtn, 6)
	self.tabBtns[tabName] = tabBtn

	local scroll = Instance.new("ScrollingFrame", self.contentArea)
	scroll.Size = UDim2.new(1, 0, 1, 0)
	scroll.BackgroundTransparency = 1
	scroll.BorderSizePixel = 0
	scroll.ScrollBarThickness = 3
	scroll.ScrollBarImageColor3 = COL_ACCENT
	scroll.CanvasSize = UDim2.new(0, 0, 0, 0)
	scroll.AutomaticCanvasSize = Enum.AutomaticSize.Y
	scroll.Visible = #self.tabs == 0
	self.tabFrames[tabName] = scroll

	local layout = Instance.new("UIListLayout", scroll)
	layout.Padding = UDim.new(0, 8)
	layout.HorizontalAlignment = Enum.HorizontalAlignment.Center
	local pad = Instance.new("UIPadding", scroll)
	pad.PaddingTop = UDim.new(0, 8)
	pad.PaddingBottom = UDim.new(0, 8)

	if #self.tabs == 0 then self.activeTab = tabName end
	table.insert(self.tabs, tabName)

	tabBtn.MouseButton1Click:Connect(function()
		for _, tn in ipairs(self.tabs) do
			self.tabFrames[tn].Visible = tn == tabName
			TweenService:Create(self.tabBtns[tn], TweenInfo.new(0.2), {
				BackgroundColor3 = tn == tabName and COL_ACCENT2 or Color3.fromRGB(30, 30, 40)
			}):Play()
		end
		self.activeTab = tabName
	end)

	local Tab = {}

	function Tab:AddButton(config)
		local bBorder = wrap(scroll, COL_ACCENT2, UDim2.new(0.88, 0, 0, 38), UDim2.new(0.06, 0, 0, 0), 11)
		local btn = Instance.new("TextButton", bBorder)
		btn.Size = UDim2.new(1, -2, 1, -2)
		btn.Position = UDim2.new(0, 1, 0, 1)
		btn.Text = config.Name or "Button"
		btn.Font = Enum.Font.Arcade
		btn.TextSize = 11
		btn.TextColor3 = Color3.new(1, 1, 1)
		btn.BackgroundColor3 = Color3.fromRGB(30, 18, 55)
		btn.BorderSizePixel = 0
		corner(btn, 10)
		btn.MouseButton1Click:Connect(function()
			TweenService:Create(btn, TweenInfo.new(0.1), { BackgroundColor3 = COL_ACCENT }):Play()
			task.delay(0.15, function()
				TweenService:Create(btn, TweenInfo.new(0.15), { BackgroundColor3 = Color3.fromRGB(30, 18, 55) }):Play()
			end)
			if config.Callback then config.Callback() end
		end)
		return btn
	end

	function Tab:AddToggle(config)
		local state = config.Default or false
		local tBorder = wrap(scroll, COL_ACCENT2, UDim2.new(0.88, 0, 0, 38), UDim2.new(0.06, 0, 0, 0), 11)
		local inner = innerFrame(tBorder, COL_BG3, 10)
		local label = Instance.new("TextLabel", inner)
		label.Size = UDim2.new(1, 0, 1, 0)
		label.Font = Enum.Font.Arcade
		label.TextSize = 11
		label.BackgroundTransparency = 1
		label.TextXAlignment = Enum.TextXAlignment.Center

		local function update()
			label.Text = (config.Name or "Toggle") .. " : " .. (state and "ON" or "OFF")
			label.TextColor3 = state and COL_ACCENT or COL_OFF
			TweenService:Create(inner, TweenInfo.new(0.15), {
				BackgroundColor3 = state and Color3.fromRGB(35, 20, 55) or COL_BG3
			}):Play()
		end
		update()

		inner.InputBegan:Connect(function(i)
			if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
				state = not state
				update()
				if config.Callback then config.Callback(state) end
			end
		end)

		return {
			Set = function(_, val)
				state = val
				update()
			end,
			Get = function() return state end
		}
	end

	function Tab:AddDropdown(config)
		local options = config.Options or {}
		local selected = config.Default or options[1] or "Select"
		local open = false
		local optH = 30

		local wrapper = Instance.new("Frame", scroll)
		wrapper.Size = UDim2.new(0.88, 0, 0, 38)
		wrapper.BackgroundTransparency = 1
		wrapper.BorderSizePixel = 0
		wrapper.ClipsDescendants = false

		local dBorder = Instance.new("Frame", wrapper)
		dBorder.Size = UDim2.new(1, 0, 0, 38)
		dBorder.BackgroundColor3 = COL_ACCENT2
		dBorder.BorderSizePixel = 0
		corner(dBorder, 11)

		local dInner = innerFrame(dBorder, Color3.fromRGB(30, 18, 55), 10)

		local dLabel = Instance.new("TextLabel", dInner)
		dLabel.Size = UDim2.new(0.75, 0, 1, 0)
		dLabel.Position = UDim2.new(0, 10, 0, 0)
		dLabel.Text = (config.Name or "Dropdown") .. ": " .. selected
		dLabel.Font = Enum.Font.Arcade
		dLabel.TextSize = 11
		dLabel.TextColor3 = COL_ACCENT
		dLabel.BackgroundTransparency = 1
		dLabel.TextXAlignment = Enum.TextXAlignment.Left

		local arrow = Instance.new("TextLabel", dInner)
		arrow.Size = UDim2.new(0, 20, 1, 0)
		arrow.Position = UDim2.new(1, -25, 0, 0)
		arrow.Text = "▼"
		arrow.Font = Enum.Font.GothamBold
		arrow.TextSize = 10
		arrow.TextColor3 = COL_ACCENT
		arrow.BackgroundTransparency = 1

		local dropList = Instance.new("Frame", wrapper)
		dropList.Position = UDim2.new(0, 1, 0, 40)
		dropList.Size = UDim2.new(1, -2, 0, #options * optH)
		dropList.BackgroundColor3 = Color3.fromRGB(18, 14, 30)
		dropList.BorderSizePixel = 0
		dropList.Visible = false
		dropList.ZIndex = 5
		corner(dropList, 8)

		for idx, opt in ipairs(options) do
			local optBtn = Instance.new("TextButton", dropList)
			optBtn.Size = UDim2.new(1, 0, 0, optH)
			optBtn.Position = UDim2.new(0, 0, 0, (idx - 1) * optH)
			optBtn.Text = opt
			optBtn.Font = Enum.Font.Arcade
			optBtn.TextSize = 10
			optBtn.TextColor3 = Color3.new(1, 1, 1)
			optBtn.BackgroundTransparency = 1
			optBtn.ZIndex = 6
			optBtn.MouseButton1Click:Connect(function()
				selected = opt
				dLabel.Text = (config.Name or "Dropdown") .. ": " .. selected
				open = false
				dropList.Visible = false
				arrow.Text = "▼"
				if config.Callback then config.Callback(selected) end
			end)
			optBtn.MouseEnter:Connect(function()
				optBtn.BackgroundColor3 = COL_ACCENT2
				TweenService:Create(optBtn, TweenInfo.new(0.1), { BackgroundTransparency = 0.7 }):Play()
			end)
			optBtn.MouseLeave:Connect(function()
				TweenService:Create(optBtn, TweenInfo.new(0.1), { BackgroundTransparency = 1 }):Play()
			end)
		end

		dInner.InputBegan:Connect(function(i)
			if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
				open = not open
				dropList.Visible = open
				arrow.Text = open and "▲" or "▼"
			end
		end)

		return {
			Get = function() return selected end,
			Set = function(_, val)
				selected = val
				dLabel.Text = (config.Name or "Dropdown") .. ": " .. selected
			end
		}
	end

	function Tab:AddInput(config)
		local iBorder = wrap(scroll, COL_ACCENT2, UDim2.new(0.88, 0, 0, 38), UDim2.new(0.06, 0, 0, 0), 11)
		local iInner = innerFrame(iBorder, Color3.fromRGB(18, 14, 30), 10)

		local box = Instance.new("TextBox", iInner)
		box.Size = UDim2.new(1, -16, 1, 0)
		box.Position = UDim2.new(0, 8, 0, 0)
		box.PlaceholderText = config.Placeholder or "Enter text..."
		box.Text = config.Default or ""
		box.Font = Enum.Font.Arcade
		box.TextSize = 11
		box.TextColor3 = Color3.new(1, 1, 1)
		box.PlaceholderColor3 = COL_OFF
		box.BackgroundTransparency = 1
		box.ClearTextOnFocus = false

		box.FocusLost:Connect(function(enter)
			if enter and config.Callback then
				config.Callback(box.Text)
			end
		end)

		return {
			Get = function() return box.Text end,
			Set = function(_, val) box.Text = val end
		}
	end

	function Tab:AddLabel(text)
		local lbl = Instance.new("TextLabel", scroll)
		lbl.Size = UDim2.new(0.88, 0, 0, 24)
		lbl.Font = Enum.Font.Arcade
		lbl.TextSize = 10
		lbl.TextColor3 = COL_ACCENT
		lbl.BackgroundTransparency = 1
		lbl.TextXAlignment = Enum.TextXAlignment.Center
		animateText(lbl, text or "")
		return lbl
	end

	return Tab
end

return GalaxyMoon
